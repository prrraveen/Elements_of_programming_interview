    if not tree:
        return None

    result = deque([[tree]])
    res = []
    while result:
        nodes = result.pop()
        children = deque()
        while nodes:
            node = nodes.pop()
            if node.left:
                children.append(node.left)
            if node.right:
                children.append(node.right)
        if children:
            result.append(children)
            res.append(list(map(lambda x: x.data, children)))
    return res
